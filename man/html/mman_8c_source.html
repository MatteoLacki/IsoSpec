30 static DWORD __map_mmap_prot_page(const int prot)
31 {
32  DWORD protect = 0;
33 
34  if (prot == PROT_NONE)
35  return protect;
36 
37  if ((prot & PROT_EXEC) != 0)
38  {
39  protect = ((prot & PROT_WRITE) != 0) ?
40  PAGE_EXECUTE_READWRITE : PAGE_EXECUTE_READ;
41  }
42  else
43  {
44  protect = ((prot & PROT_WRITE) != 0) ?
45  PAGE_READWRITE : PAGE_READONLY;
46  }
47 
48  return protect;
49 }

51 static DWORD __map_mmap_prot_file(const int prot)
52 {
53  DWORD desiredAccess = 0;
54 
55  if (prot == PROT_NONE)
56  return desiredAccess;
57 
58  if ((prot & PROT_READ) != 0)
59  desiredAccess |= FILE_MAP_READ;
60  if ((prot & PROT_WRITE) != 0)
61  desiredAccess |= FILE_MAP_WRITE;
62  if ((prot & PROT_EXEC) != 0)
63  desiredAccess |= FILE_MAP_EXECUTE;
64 
65  return desiredAccess;
66 }

68 void* mmap(void *addr, size_t len, int prot, int flags, int fildes, OffsetType off)
69 {
70  HANDLE fm, h;
71 
72  void * map = MAP_FAILED;
73 
74 #ifdef _MSC_VER
75 #pragma warning(push)
76 #pragma warning(disable: 4293)
77 #endif
78 
79  const DWORD dwFileOffsetLow = (sizeof(OffsetType) <= sizeof(DWORD)) ?
80  (DWORD)off : (DWORD)(off & 0xFFFFFFFFL);
81  const DWORD dwFileOffsetHigh = (sizeof(OffsetType) <= sizeof(DWORD)) ?
82  (DWORD)0 : (DWORD)((off >> 32) & 0xFFFFFFFFL);
83  const DWORD protect = __map_mmap_prot_page(prot);
84  const DWORD desiredAccess = __map_mmap_prot_file(prot);
85 
86  const OffsetType maxSize = off + (OffsetType)len;
87 
88  const DWORD dwMaxSizeLow = (sizeof(OffsetType) <= sizeof(DWORD)) ?
89  (DWORD)maxSize : (DWORD)(maxSize & 0xFFFFFFFFL);
90  const DWORD dwMaxSizeHigh = (sizeof(OffsetType) <= sizeof(DWORD)) ?
91  (DWORD)0 : (DWORD)((maxSize >> 32) & 0xFFFFFFFFL);
92 
93 #ifdef _MSC_VER
94 #pragma warning(pop)
95 #endif
96 
97  errno = 0;
98 
99  if (len == 0
100  /* Unsupported flag combinations */
101  || (flags & MAP_FIXED) != 0
102  /* Usupported protection combinations */
103  || prot == PROT_EXEC)
104  {
105  errno = EINVAL;
106  return MAP_FAILED;
107  }
108 
109  h = ((flags & MAP_ANONYMOUS) == 0) ?
110  (HANDLE)_get_osfhandle(fildes) : INVALID_HANDLE_VALUE;
111 
112  if ((flags & MAP_ANONYMOUS) == 0 && h == INVALID_HANDLE_VALUE)
113  {
114  errno = EBADF;
115  return MAP_FAILED;
116  }
117 
118  fm = CreateFileMapping(h, NULL, protect, dwMaxSizeHigh, dwMaxSizeLow, NULL);
119 
120  if (fm == NULL)
121  {
122  errno = __map_mman_error(GetLastError(), EPERM);
123  return MAP_FAILED;
124  }
125 
126  map = MapViewOfFile(fm, desiredAccess, dwFileOffsetHigh, dwFileOffsetLow, len);
127 
128  CloseHandle(fm);
129 
130  if (map == NULL)
131  {
132  errno = __map_mman_error(GetLastError(), EPERM);
133  return MAP_FAILED;
134  }
135 
136  return map;
137 }
