160 unsigned int parse_formula(const char* formula, std::vector<const double*>& isotope_masses, std::vector<const double*>& isotope_probabilities, int** isotopeNumbers, int** atomCounts, unsigned int* confSize)
161 {
162  // This function is NOT guaranteed to be secure against malicious input. It should be used only for debugging.
163  size_t slen = strlen(formula);
164  // Yes, it would be more elegant to use std::string here, but it's the only promiment place where it would be used in IsoSpec, and avoiding it here
165  // means we can run the whole thing through Clang's memory sanitizer without the need for instrumented libc++/libstdc++. That's worth messing with char pointers a
166  // little bit.
167  std::vector<std::pair<const char*, size_t> > elements;
168  std::vector<int> numbers;
169 
170  if(slen == 0)
171  throw invalid_argument("Invalid formula: can't be empty");
172 
173  if(!isdigit(formula[slen-1]))
174  throw invalid_argument("Invalid formula: every element must be followed by a number - write H2O1 and not H2O for water");
175 
176  for(size_t ii=0; ii<slen; ii++)
177  if(!isdigit(formula[ii]) && !isalpha(formula[ii]))
178  throw invalid_argument("Ivalid formula: contains invalid (non-digit, non-alpha) character");
179 
180  size_t position = 0;
181  size_t elem_end = 0;
182  size_t digit_end = 0;
183 
184  while(position < slen)
185  {
186  elem_end = position;
187  while(isalpha(formula[elem_end]))
188  elem_end++;
189  digit_end = elem_end;
190  while(isdigit(formula[digit_end]))
191  digit_end++;
192  elements.emplace_back(&formula[position], elem_end-position);
193  numbers.push_back(atoi(&formula[elem_end]));
194  position = digit_end;
195  }
196 
197  std::vector<int> element_indexes;
198 
199  for (unsigned int i=0; i<elements.size(); i++)
200  {
201  int idx = -1;
202  for(int j=0; j<ISOSPEC_NUMBER_OF_ISOTOPIC_ENTRIES; j++)
203  {
204  if ((strlen(elem_table_symbol[j]) == elements[i].second) && (strncmp(elements[i].first, elem_table_symbol[j], elements[i].second) == 0))
205  {
206  idx = j;
207  break;
208  }
209  }
210  if(idx < 0)
211  throw invalid_argument("Invalid formula");
212  element_indexes.push_back(idx);
213  }
214 
215  vector<int> _isotope_numbers;
216 
217  for(vector<int>::iterator it = element_indexes.begin(); it != element_indexes.end(); ++it)
218  {
219  int num = 0;
220  int at_idx = *it;
221  int atomicNo = elem_table_atomicNo[at_idx];
222  while(at_idx < ISOSPEC_NUMBER_OF_ISOTOPIC_ENTRIES && elem_table_atomicNo[at_idx] == atomicNo)
223  {
224  at_idx++;
225  num++;
226  }
227  _isotope_numbers.push_back(num);
228  }
229 
230  for(vector<int>::iterator it = element_indexes.begin(); it != element_indexes.end(); ++it)
231  {
232  isotope_masses.push_back(&elem_table_mass[*it]);
233  isotope_probabilities.push_back(&elem_table_probability[*it]);
234  };
235 
236  const unsigned int dimNumber = elements.size();
237 
238  *isotopeNumbers = array_copy<int>(_isotope_numbers.data(), dimNumber);
239  *atomCounts = array_copy<int>(numbers.data(), dimNumber);
240  *confSize = dimNumber * sizeof(int);
241 
242  return dimNumber;
243 
244 }

286 IsoThresholdGenerator::IsoThresholdGenerator(Iso&& iso, double _threshold, bool _absolute, int tabSize, int hashSize, bool reorder_marginals)
287 : IsoGenerator(std::move(iso)),
288 Lcutoff(_threshold <= 0.0 ? std::numeric_limits<double>::lowest() : (_absolute ? log(_threshold) : log(_threshold) + modeLProb))
289 {
290  counter = new int[dimNumber];
291  maxConfsLPSum = new double[dimNumber-1];
292  marginalResultsUnsorted = new PrecalculatedMarginal*[dimNumber];
293 
294  empty = false;
295 
296  for(int ii=0; ii<dimNumber; ii++)
297  {
298  counter[ii] = 0;
299  marginalResultsUnsorted[ii] = new PrecalculatedMarginal(std::move(*(marginals[ii])),
300  Lcutoff - modeLProb + marginals[ii]->getModeLProb(),
301  true,
302  tabSize,
303  hashSize);
304 
305  if(!marginalResultsUnsorted[ii]->inRange(0))
306  empty = true;
307  }
308 
309  if(reorder_marginals)
310  {
311  OrderMarginalsBySizeDecresing comparator(marginalResultsUnsorted);
312  int* tmpMarginalOrder = new int[dimNumber];
313 
314  for(int ii=0; ii<dimNumber; ii++)
315  tmpMarginalOrder[ii] = ii;
316 
317  std::sort(tmpMarginalOrder, tmpMarginalOrder + dimNumber, comparator);
318  marginalResults = new PrecalculatedMarginal*[dimNumber];
319 
320  for(int ii=0; ii<dimNumber; ii++)
321  marginalResults[ii] = marginalResultsUnsorted[tmpMarginalOrder[ii]];
322 
323  marginalOrder = new int[dimNumber];
324  for(int ii = 0; ii<dimNumber; ii++)
325  marginalOrder[tmpMarginalOrder[ii]] = ii;
326 
327  delete[] tmpMarginalOrder;
328 
329  }
330  else
331  {
332  marginalResults = marginalResultsUnsorted;
333  marginalOrder = nullptr;
334  }
335 
336  lProbs_ptr_start = marginalResults[0]->get_lProbs_ptr();
337 
338  if(dimNumber > 1)
339  maxConfsLPSum[0] = marginalResults[0]->getModeLProb();
340 
341  for(int ii=1; ii<dimNumber-1; ii++)
342  maxConfsLPSum[ii] = maxConfsLPSum[ii-1] + marginalResults[ii]->getModeLProb();
343 
344  lProbs_ptr = lProbs_ptr_start;
345 
346  partialLProbs_second = partialLProbs;
347  partialLProbs_second++;
348 
349  if(!empty)
350  {
351  recalc(dimNumber-1);
352  counter[0]--;
353  lProbs_ptr--;
354  }
355  else
356  {
357  terminate_search();
358  lcfmsv = std::numeric_limits<double>::infinity();
359  }

461 bool IsoOrderedGenerator::advanceToNextConfiguration()
462 {
463  if(pq.size() < 1)
464  return false;
465 
466 
467  topConf = pq.top();
468  pq.pop();
469 
470  int* topConfIsoCounts = getConf(topConf);
471 
472  currentLProb = *(reinterpret_cast<double*>(topConf));
473  currentMass = combinedSum( topConfIsoCounts, masses, dimNumber );
474  currentProb = exp(currentLProb);
475 
476  ccount = -1;
477  for(int j = 0; j < dimNumber; ++j)
478  {
479  if(marginalResults[j]->probeConfigurationIdx(topConfIsoCounts[j] + 1))
480  {
481  if(ccount == -1)
482  {
483  topConfIsoCounts[j]++;
484  *(reinterpret_cast<double*>(topConf)) = combinedSum(topConfIsoCounts, logProbs, dimNumber);
485  pq.push(topConf);
486  topConfIsoCounts[j]--;
487  ccount = j;
488  }
489  else
490  {
491  void* acceptedCandidate = allocator.newConf();
492  int* acceptedCandidateIsoCounts = getConf(acceptedCandidate);
493  memcpy(acceptedCandidateIsoCounts, topConfIsoCounts, confSize);
494 
495  acceptedCandidateIsoCounts[j]++;
496 
497  *(reinterpret_cast<double*>(acceptedCandidate)) = combinedSum(acceptedCandidateIsoCounts, logProbs, dimNumber);
498 
499  pq.push(acceptedCandidate);
500  }
501  }
502  if(topConfIsoCounts[j] > 0)
503  break;
504  }
505  if(ccount >=0)
506  topConfIsoCounts[ccount]++;
507 
508  return true;
509 }

615 bool IsoLayeredGenerator::advanceToNextLayer()
616 {
617  layers += 1;
618  double maxFringeLprob = -std::numeric_limits<double>::infinity();
619 
620  if(current == nullptr)
621  return false;
622  int accepted_in_this_layer = 0;
623  Summator prob_in_this_layer(totalProb);
624 
625  void* topConf;
626 
627  while(current->size() > 0)
628  {
629  topConf = current->back();
630  current->pop_back();
631 
632  double top_lprob = getLProb(topConf);
633 
634  if(top_lprob >= lprobThr)
635  {
636  newaccepted.push_back(topConf);
637  accepted_in_this_layer++;
638  prob_in_this_layer.add(exp(top_lprob));
639  }
640  else
641  {
642  next->push_back(topConf);
643  continue;
644  }
645 
646  int* topConfIsoCounts = getConf(topConf);
647 
648  for(int j = 0; j < dimNumber; ++j)
649  {
650  // candidate cannot refer to a position that is
651  // out of range of the stored marginal distribution.
652  if(marginalResults[j]->probeConfigurationIdx(topConfIsoCounts[j] + 1))
653  {
654  memcpy(candidate, topConfIsoCounts, confSize);
655  candidate[j]++;
656 
657  void* acceptedCandidate = allocator.newConf();
658  int* acceptedCandidateIsoCounts = getConf(acceptedCandidate);
659  memcpy( acceptedCandidateIsoCounts, candidate, confSize);
660 
661  double newConfProb = combinedSum(
662  candidate,
663  logProbs,
664  dimNumber
665  );
666 
667 
668 
669  *(reinterpret_cast<double*>(acceptedCandidate)) = newConfProb;
670 
671  if(newConfProb >= lprobThr)
672  current->push_back(acceptedCandidate);
673  else
674  {
675  next->push_back(acceptedCandidate);
676  if(newConfProb > maxFringeLprob)
677  maxFringeLprob = top_lprob;
678  }
679  }
680  if(topConfIsoCounts[j] > 0)
681  break;
682  }
683  }
684 
685  if(next == nullptr || next->size() < 1)
686  return false;
687  else
688  {
689  if(prob_in_this_layer.get() < targetCoverage)
690  {
691  std::vector<void*>* nnew = current;
692  nnew->clear();
693  current = next;
694  next = nnew;
695  int howmany = floor(current->size()*percentageToExpand);
696  lprobThr = getLProb(quickselect(current->data(), howmany, 0, current->size()));
697  totalProb = prob_in_this_layer;
698  }
699  else
700  {
701  delete next;
702  next = nullptr;
703  delete current;
704  current = nullptr;
705  int start = 0;
706  int end = accepted_in_this_layer - 1;
707  void* swapspace;
708 
709  if(do_trim)
710  {
711  void** lastLayer = &(newaccepted.data()[newaccepted.size()-accepted_in_this_layer]);
712 
713  Summator qsprob(totalProb);
714  while(totalProb.get() < targetCoverage)
715  {
716  if(start == end)
717  break;
718 
719  // Partition part
720 
721  int len = end - start;
722 #ifdef BUILDING_R
723  int pivot = len/2 + start; // We're very definitely NOT switching to R to use a RNG, and if R sees us use C RNG it complains...
724 #else
725  int pivot = rand() % len + start;
726 #endif
727  void* pval = lastLayer[pivot];
728  double pprob = getLProb(pval);
729  mswap(lastLayer[pivot], lastLayer[end-1]);
730  int loweridx = start;
731  for(int i=start; i<end-1; i++)
732  {
733  if(getLProb(lastLayer[i]) > pprob)
734  {
735  mswap(lastLayer[i], lastLayer[loweridx]);
736  loweridx++;
737  }
738  }
739  mswap(lastLayer[end-1], lastLayer[loweridx]);
740 
741  // Selection part
742
