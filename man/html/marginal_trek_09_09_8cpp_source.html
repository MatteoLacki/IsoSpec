55 Conf initialConfigure(const int atomCnt, const int isotopeNo, const double* probs, const double* lprobs)
56 {
61  Conf res = new int[isotopeNo];
62 
63  // This approximates the mode (heuristics: the mean is close to the mode).
64  for(int i = 0; i < isotopeNo; ++i )
65  res[i] = int( atomCnt * probs[i] ) + 1;
66 
67  // The number of assigned atoms above.
68  int s = 0;
69 
70  for(int i = 0; i < isotopeNo; ++i) s += res[i];
71 
72  int diff = atomCnt - s;
73 
74  // Too little: enlarging fist index.
75  if( diff > 0 ){
76  res[0] += diff;
77  }
78  // Too much: trying to redistribute the difference: hopefully the first element is the largest.
79  if( diff < 0 ){
80  diff = abs(diff);
81  int i = 0, coordDiff = 0;
82 
83  while( diff > 0){
84  coordDiff = res[i] - diff;
85 
86  if( coordDiff >= 0 ){
87  res[i] -= diff;
88  diff = 0;
89  } else {
90  res[i] = 0;
91  i++;
92  diff = abs(coordDiff);
93  }
94  }
95  }
96 
97  // What we computed so far will be very close to the mode: hillclimb the rest of the way
98 
99  bool modified = true;
100  double LP = unnormalized_logProb(res, lprobs, isotopeNo);
101  double NLP;
102 
103  while(modified)
104  {
105  modified = false;
106  for(int ii = 0; ii<isotopeNo; ii++)
107  for(int jj = 0; jj<isotopeNo; jj++)
108  if(ii != jj && res[ii] > 0)
109  {
110  res[ii]--;
111  res[jj]++;
112  NLP = unnormalized_logProb(res, lprobs, isotopeNo);
113  if(NLP>LP || (NLP==LP && ii>jj))
114  {
115  modified = true;
116  LP = NLP;
117  }
118  else
119  {
120  res[ii]++;
121  res[jj]--;
122  }
123  }
124 
125 
126  }
127  return res;
128 }

150 double* getMLogProbs(const double* probs, int isoNo)
151 {
157  int curr_method = fegetround();
158  fesetround(FE_UPWARD);
159  double* ret = new double[isoNo];
160 
161  // here we change the table of probabilities and log it.
162  for(int i = 0; i < isoNo; i++)
163  {
164  ret[i] = log(probs[i]);
165  for(int j=0; j<ISOSPEC_NUMBER_OF_ISOTOPIC_ENTRIES; j++)
166  if(elem_table_probability[j] == probs[i])
167  {
168  ret[i] = elem_table_log_probability[j];
169  break;
170  }
171  }
172  fesetround(curr_method);
173  return ret;
174 }

285 bool MarginalTrek::add_next_conf()
286 {
291  if(pq.size() < 1) return false;
292 
293  Conf topConf = pq.top();
294  pq.pop();
295  ++current_count;
296  visited[topConf] = current_count;
297 
298  _confs.push_back(topConf);
299  _conf_masses.push_back(mass(topConf, atom_masses, isotopeNo));
300  double logprob = logProb(topConf);
301  _conf_lprobs.push_back(logprob);
302 
303 
304  totalProb.add( exp( logprob ) );
305 
306  for( unsigned int i = 0; i < isotopeNo; ++i )
307  {
308  for( unsigned int j = 0; j < isotopeNo; ++j )
309  {
310  // Growing index different than decreasing one AND
311  // Remain on simplex condition.
312  if( i != j && topConf[j] > 0 ){
313  copyConf(topConf, candidate, isotopeNo);
314 
315  ++candidate[i];
316  --candidate[j];
317 
318  // candidate should not have been already visited.
319  if( visited.count( candidate ) == 0 )
320  {
321  Conf acceptedCandidate = allocator.makeCopy(candidate);
322  pq.push(acceptedCandidate);
323 
324  visited[acceptedCandidate] = 0;
325  }
326  }
327  }
328  }
329 
330  return true;
331 }

333 int MarginalTrek::processUntilCutoff(double cutoff)
334 {
335  Summator s;
336  int last_idx = -1;
337  for(unsigned int i=0; i<_conf_lprobs.size(); i++)
338  {
339  s.add(_conf_lprobs[i]);
340  if(s.get() >= cutoff)
341  {
342  last_idx = i;
343  break;
344  }
345  }
346  if(last_idx > -1)
347  return last_idx;
348 
349  while(totalProb.get() < cutoff && add_next_conf()) {}
350  return _conf_lprobs.size();
351 }

362 PrecalculatedMarginal::PrecalculatedMarginal(Marginal&& m,
363  double lCutOff,
364  bool sort,
365  int tabSize,
366  int hashSize
367 ) : Marginal(std::move(m)),
368 allocator(isotopeNo, tabSize)
369 {
370  const ConfEqual equalizer(isotopeNo);
371  const KeyHasher keyHasher(isotopeNo);
372  const ConfOrderMarginalDescending orderMarginal(atom_lProbs, isotopeNo);
373 
374  std::unordered_set<Conf,KeyHasher,ConfEqual> visited(hashSize,keyHasher,equalizer);
375 
376  Conf currentConf = allocator.makeCopy(mode_conf);
377  if(logProb(currentConf) >= lCutOff)
378  {
379  // create a copy and store a ptr to the *same* copy in both structures
380  // (save some space and time)
381  auto tmp = allocator.makeCopy(currentConf);
382  configurations.push_back(tmp);
383  visited.insert(tmp);
384  }
385 
386  unsigned int idx = 0;
387 
388  while(idx < configurations.size())
389  {
390  memcpy(currentConf, configurations[idx], sizeof(int)*isotopeNo);
391  idx++;
392  for(unsigned int ii = 0; ii < isotopeNo; ii++ )
393  for(unsigned int jj = 0; jj < isotopeNo; jj++ )
394  if( ii != jj && currentConf[jj] > 0)
395  {
396  currentConf[ii]++;
397  currentConf[jj]--;
398 
399  if (visited.count(currentConf) == 0 && logProb(currentConf) >= lCutOff)
400  {
401  // create a copy and store a ptr to the *same* copy in
402  // both structures (save some space and time)
403  auto tmp = allocator.makeCopy(currentConf);
404  visited.insert(tmp);
405  configurations.push_back(tmp);
406  // std::cout << " V: "; for (auto it : visited) std::cout << it << " "; std::cout << std::endl;
407  }
408 
409  currentConf[ii]--;
410  currentConf[jj]++;
411 
412  }
413  }
414 
415  // orderMarginal defines the order of configurations (compares their logprobs)
416  // akin to key in Python sort.
417  if(sort)
418  std::sort(configurations.begin(), configurations.end(), orderMarginal);
419 
420 
421  confs = &configurations[0];
422  no_confs = configurations.size();
423  lProbs = new double[no_confs+1];
424  eProbs = new double[no_confs];
425  masses = new double[no_confs];
426 
427 
428  for(unsigned int ii=0; ii < no_confs; ii++)
429  {
430  lProbs[ii] = logProb(confs[ii]);
431  eProbs[ii] = exp(lProbs[ii]);
432  masses[ii] = mass(confs[ii], atom_masses, isotopeNo);
433  }
434  lProbs[no_confs] = -std::numeric_limits<double>::infinity();
435 }
